export INPUT_DIR="./datasets/nuscenes/cam_front/test"
export OUTPUT_DIR="./outputs/cam_front/test_01"
export USER_PROMPT="Please analyze the following road surveillance video and generate a dense caption describing only the real, visible events in 1–3 sentences. Focus on elements such as: - Vehicles, pedestrians, infrastructure (e.g., buildings, traffic lights, speed bumps) - Actions (e.g., stopping, turning, walking, waiting) - Environmental characteristics (e.g., road markings, lighting conditions) Do not guess or hallucinate. Do not describe intentions, emotions, or unseen context. Avoid speculation. Only describe what is clearly observable." 
export SYSTEM_PROMPT="You are a helpful video analyzer and precise vision-language model trained to analyze videos. Your task is to generate dense, concise, and grounded descriptions of visual content in videos. You must focus only on visible events, objects, and actions."
./experimental/afasale/cosmos-reason1-scripts/inference/batch_inference.py \
--prompt $USER_PROMPT \
--system_prompt $SYSTEM_PROMPT \
--video-dir $INPUT_DIR \
--output-dir $OUTPUT_DIR \
--copy-videos \
--json \
--fps 16 \
--model nvidia/Cosmos-Reason1-7B
